87 void TPM1_IRQHandler(void) {
88  /* Ch1 ISR */
89  if (TPM1->CONTROLS[1].CnSC & TPM_CnSC_CHF_MASK) {
90  FPTE->PCOR |= (1 << 21); /* Turn off trigger pin*/
91  TPM1->CONTROLS[1].CnSC &= ~TPM_CnSC_CHIE_MASK; /* Disable TMP1_Ch1 interupts */
92  TPM1->CONTROLS[1].CnSC |= TPM_CnSC_CHF_MASK; /* Clear TMP1_Ch1 flag */
93  }
94 
95  /* Ch0 ISR */
96  if (TPM1->CONTROLS[0].CnSC & TPM_CnSC_CHF_MASK) {
97  if ( SonarState == SONAR_TRIGGER_SENT ) {
98  TPM1->CNT = 0; /* Reset counter */
99  SonarState = SONAR_CAPTURE_START; /* Change sonar state to CAPTURE_START */
100  } else {
101  TPM1->SC |= TPM_SC_TOF_MASK; /* Clear TPM1 Overflow flag */
102  TPM1->SC &= ~TPM_SC_TOIE_MASK; /* Disable TPM1 Overflow interupt */
103  SonarState = SONAR_CAPTURE_END; /* Change sonar state to CAPTURE_END */
104  SonarDistHandler(TPM1->CONTROLS[0].CnV/SONAR_TICKS_PER_CM); /* Execute user results handler */
105  SonarState = SONAR_IDLE; /* Change sonar state to CAPTURE_END */
106  success++;
107  }
108  TPM1->CONTROLS[0].CnSC |= TPM_CnSC_CHF_MASK; /* clear TMP1_Ch0 flag */
109  }
110 
111  /* TPM1 overflow handler */
112  if ((TPM1->SC & TPM_SC_TOF_MASK ) && (TPM1->SC & TPM_SC_TOIE_MASK )) {
113  TPM1->SC |= TPM_SC_TOF_MASK; /* Clear TPM1 Overflow flag */
114  TPM1->SC &= ~TPM_SC_TOIE_MASK; /* Disable TPM1 Overflow interupt */
115  SonarState = SONAR_CAPTURE_OVERFLOW; /* set Sonar to CAPTURE_OVERFLOW state */
116  SonarDistHandler(0u); /* Execute user results handler */
117  fail++;
118  }
119 }

133 void PIT_IRQHandler(void){
134  /* CH1 ISR */
135  if (PIT->CHANNEL[0].TFLG & PIT_TFLG_TIF_MASK) {
136  if (SonarMode == CONTINUOUS) {
137  switch(SonarState) {
138  case SONAR_IDLE:
139  if (ServoMode == SWEEP && ServoState == IDLE) Servo_sweep_step();
140  SendTrigger();
141  break;
142  case SONAR_CAPTURE_OVERFLOW:
143  SendTrigger();
144  break;
145  case SONAR_CAPTURE_END:
146  if (ServoMode == SWEEP && ServoState == IDLE) Servo_sweep_step();
147  SendTrigger();
148  break;
149  default:
150  break;
151  }
152  }
153  PIT->CHANNEL[0].TFLG |= PIT_TFLG_TIF_MASK; /* Clear Interupt Flag */
154  } else {
155  /* CH2 ISR */
156  /* Servo reached its destination. Stop countdown */
157  ServoState = IDLE; /* Set sonar state to idle */
158  PIT->CHANNEL[1].TCTRL &= ~PIT_TCTRL_TEN_MASK; /* Disable timer */
159  PIT->CHANNEL[1].TFLG |= PIT_TFLG_TIF_MASK; /* Clear Interupt Flag */
160  }
161 }
